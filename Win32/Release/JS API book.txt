/**
提交表单：
fieldsJsonStr的格式为：
'[
	{
		"name" : "field1", 
		"value" : "text1", 
		"type" : 0 //0为普通字段（可选）
	 },

	{
		"type" : 1, //1为文件字段（必选）
		"name" : "file", 
		"value" : "C:/1.txt", //文件路径
		"fileName" : "1.txt" //文件名（可选）
	},
]' ---- 注意：这是一个字符串
*/
